```python
# ==========================================
# 正则军火库：物理级前置拦截器
# ==========================================
raw_filename = file_name or file_path.split('\\')[-1].split('/')[-1]
cleaned_filename = raw_filename

# 1. 获取系统正则规则
regex_rules_str = db.get_config("filename_clean_regex", "")

# 2. 逐行解析并应用正则
if regex_rules_str:
    for line in regex_rules_str.splitlines():
        rule = line.strip()
        # 跳过空行和注释行
        if not rule or rule.startswith('#'):
            continue
        try:
            # 忽略大小写，执行全局替换为空格
            cleaned_filename = re.sub(rule, ' ', cleaned_filename, flags=re.IGNORECASE)
        except re.error as e:
            logger.warning(f"[RegexLab] 无效的正则规则跳过: {rule}, 错误: {e}")

# 3. 基础标点符号清理
cleaned_filename = re.sub(r'[._\[\]\(\)]+', ' ', cleaned_filename)
cleaned_filename = ' '.join(cleaned_filename.split()).strip()

# 4. 兜底策略
if not cleaned_filename:
    cleaned_filename = raw_filename

logger.info(f"[RegexLab] 物理正则去噪完成: '{raw_filename}' -> '{cleaned_filename}'")

# 将清洗后的文件名传给 AI
logger.info(f"[AI] 调用 AI Agent 分析文件名: {cleaned_filename}")
ai_result = loop.run_until_complete(ai_agent.ai_identify_media(
    cleaned_name=cleaned_filename,  # 使用清洗后的文件名
    full_path=file_path,
    type_hint=media_type
))
```
